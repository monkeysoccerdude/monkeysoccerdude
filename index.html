var Blocksx = [],Blocksy = [];
var Fastblocksx = [],Fastblocksy = [];
var Coinsx = [],Coinsy = [],SCoinsx = [],SCoinsy = [];
var time=0,coin=0,time3=0,time4=0;
//you need to avoid the blocks and collect the coins!
var drawBlock = function(){
    fill(0, 0, 255);
    for(var i=0;i=Blocksx[i]-15&&X<=Blocksx[i]+40&&Y>=Blocksy[i]-10&&Y<=Blocksy[i]+40){
            return true;
        }
        if(X<93||X>310){
            return true;
        }
    }
    for(var i=0;i=Fastblocksx[i]-10&&X<=Fastblocksx[i]+40&&Y>=Fastblocksy[i]-10&&Y<=Fastblocksy[i]+40){
            return true;
        }
        if(X<90||X>310){
            return true;
        }
    }
    return false;
};
var IfWinstonGetsCoin = function(){
    var ans = 0;
    for(var i=0;i=Coinsx[i]-28&&X<=Coinsx[i]+50&&Y>=Coinsy[i]-20&&Y<=Coinsy[i]+50){
            Coinsx[i]=-3;
            ans++;
        }
    }
    for(var i=0;i=SCoinsx[i]-10&&X<=SCoinsx[i]+40&&Y>=SCoinsy[i]-10&&Y<=SCoinsy[i]+40){
            SCoinsx[i]=-50;
            scoins++;
        }
    }
    return ans;
};
var dead=false;

        time3=0;
        Fastblocksx.push(random(100,270));
        Fastblocksy.push(0);
    }
    if(time4>=500+random(0,10)){
        time4-=500;
        SCoinsx.push(random(100,270));
        SCoinsy.push(0);
    }
    for(var i=0;i=50){
            text("Coin Master",142,259);

    text("your coins:"+coin,10,45);
    time++;
    time2++;
    time3++;
    time4++;
    if(dead===false){
        score++;
    }
};
